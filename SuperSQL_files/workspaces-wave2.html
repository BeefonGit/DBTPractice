select orders.id,
orders.user_id,
orders.order_date,
orders.status,
customers.first_name,
customers.last_name,
payment.paymentmethod,
payment.status,
payment.amount
from raw.jaffle_shop.orders
inner join raw.jaffle_shop.customers on orders.user_id = customers.id
inner join raw.stripe.payment on orders.id = payment.orderid
where payment.amount > 1000
and payment.paymentmethod = 'credit_card'
order by payment.amount desc, order.order_date
